# Customize Scroller



```js
niceScrollOpts: {  // niceScroll options
    cursorcolor: "none",
    cursorwidth: 10,
    scrollspeed: 60,
    cursorborderradius: 0,
    mousescrollstep: 40,
    background: "none",
    cursorborder: "none",
    autohidemode: true,
    boxzoom: false,
    zindex: 990
}
, scrollTarget: '.scroll_area'  // scroll area selector
, mode: 'horizontal'  // 'vertical', 'horizontal'
, btnClickDistance: 7  // percent(%)
```
